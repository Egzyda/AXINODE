function updateBattle(state: GameState, deltaTime: number): GameState {
  const battle = state.currentBattle!;
  
  // 10ç§’ã”ã¨ã«å‡¦ç†
  if (battle.elapsedTime % 10 < deltaTime) {
    // æˆ¦é—˜åŠ›è¨ˆç®—
    const playerPower = calculateCombatPower(battle.playerForces);
    const enemyPower = calculateCombatPower(battle.enemyForces);
    
    // ãƒ€ãƒ¡ãƒ¼ã‚¸åˆ¤å®š
    const playerDamage = enemyPower * 0.1;
    const enemyDamage = playerPower * 0.08;
    
    battle.playerForces.total -= Math.floor(playerDamage);
    battle.enemyForces.total -= Math.floor(enemyDamage);
    
    // å£«æ°—å¤‰å‹•
    if (playerPower > enemyPower) {
      battle.playerForces.morale = Math.min(100, battle.playerForces.morale + 2);
      battle.enemyForces.morale = Math.max(0, battle.enemyForces.morale - 3);
    } else {
      battle.playerForces.morale = Math.max(0, battle.playerForces.morale - 3);
      battle.enemyForces.morale = Math.min(100, battle.enemyForces.morale + 2);
    }
    
    // å‹æ•—åˆ¤å®š
    if (battle.enemyForces.total <= 0 || battle.enemyForces.morale <= 0) {
      battle.result = 'victory';
      state = handleVictory(state, battle);
    } else if (battle.playerForces.total <= 0 || battle.playerForces.morale <= 0) {
      battle.result = 'defeat';
      state = handleDefeat(state, battle);
    }
  }
  
  battle.elapsedTime += deltaTime;
  return state;
}

function calculateCombatPower(forces: any): number {
  const equipmentCoef = forces.equipmentRate >= 80 ? 0.9 : 
                        forces.equipmentRate >= 60 ? 0.75 : 0.5;
  
  const moraleCoef = forces.morale >= 80 ? 1.0 : 
                     forces.morale >= 60 ? 0.85 : 0.65;
  
  return forces.total * equipmentCoef * moraleCoef * (forces.techBonus || 1.0);
}
